; EQU things, rings, and strings.
; Using port J, pin 0, 


PORTJ		EQU		$28		;PORTJ ADDR
DDRJ		EQU		$29		;DDRJ ADDR
INMASK		EQU		$00		;SET PORT FOR INPUT
PIN_0		EQU		$00
;BUTTON
			LDAA		#INMASK		; CONFIGURE FOR INPUT.
			STAA		DDRJ		; SET AS INPUT.
			LDAA		#PIN_0		; CONFIGURE USE OF PIN 0.
			STAA		PORTJ		; SET PIN 0 AS INPUT.
			


			LDAB		#$01		; LOAD ACC ‘B’ WITH VALUE 1
WAITING		LDAA		PORTJ		; THIS SHOULD READ BTN STATE
			CBA						; A - B, IF BTN_PRESSED == TRUE,
			BNE			WAITING		;	SHOULD EQUAL 0

; CONTINUE TO HERE WHEN PRESSED, PRINT OUT MSG AND ;BEEP
			LDD		#BTN_PSHD
			LDX		PRINTF
			JSR		0,X
